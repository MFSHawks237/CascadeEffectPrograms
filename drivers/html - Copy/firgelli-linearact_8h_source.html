00064 // Task to control motor A
00065 task _FLACcontrolTaskA() {
00066   long _currentEncVal;                // _currentEncVal encoder value
00067   bool _reverse = false;
00068   bool _done = false;
00069   long _lastEncoderCount = 0;
00070   long _stallCounter = 0;
00071 
00072   // This has to be done to prevent the PID regulator from
00073   // messing with the motor speeds
00074   nMotorPIDSpeedCtrl[motorA] = mtrNoReg;
00075 
00076   // we're not stalled just yet, so reset the variable.
00077   _stalled[motorA] = false;
00078 
00079   _currentEncVal = nMotorEncoder[motorA];
00080   _stallCounter = 0;
00081 
00082   // Flip it and reverse it
00083   if (_encoderTarget[motorA] < _currentEncVal)
00084     _reverse = true;
00085 
00086   while (!_done) {
00087     _currentEncVal = nMotorEncoder[motorA];
00088 
00089     // Are we there yet?
00090     if (_reverse && (_currentEncVal <= _encoderTarget[motorA]))
00091       _done = true;
00092     else if (!_reverse && (_currentEncVal >= _encoderTarget[motorA]))
00093       _done = true;
00094 
00095     // Stall detection magic bits happening here.
00096     if (!_done && (abs(_lastEncoderCount - _currentEncVal) > 0)) {
00097       _stallCounter = 0;
00098     } else if (!_done && (_stallCounter > STALL_TIME_SLOW) && _motorHighPower[motorA] < 50) {
00099       _stalled[motorA] = true;
00100       _done = true;
00101     } else if (!_done && (_stallCounter > STALL_TIME_SLOW) && _motorHighPower[motorA] >= 50) {
00102       _stalled[motorA] = true;
00103       _done = true;
00104     } else {
00105       _stallCounter++;
00106     }
00107 
00108     // Only change the motor speed if we're not done yet
00109     if (!_done)
00110       motor[motorA] = (_reverse) ? -_motorHighPower[motorA] : _motorHighPower[motorA];
00111 
00112     // update the _lastEncoderCount for the stall detection
00113     _lastEncoderCount = _currentEncVal;
00114     EndTimeSlice();
00115   }
00116   motor[motorA] = 0; //turn motor off
00117 }

00240 void _FLACcontrolTasks(tMotor _motor, int _highPower, long _encTarget) {
00241   switch(_motor) {
